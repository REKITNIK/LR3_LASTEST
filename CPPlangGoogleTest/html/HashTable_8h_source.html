233template<typename K, typename V>
234HashTable<K, V>& HashTable<K, V>::operator=(const HashTable& other) {
235 if (this != &other) {
236 // Идиома Copy-and-Swap для гарантии безопасности исключений.
237 // 1. Создаем временную копию. Если здесь произойдет исключение (bad_alloc),
238 // текущий объект (this) останется нетронутым.
239 HashTable temp(other);
240
241 // 2. Меняем местами внутренние ресурсы this и временного объекта.
242 // Операции swap не выбрасывают исключений.
243 std::swap(buckets, temp.buckets);
244 std::swap(bucket_count, temp.bucket_count);
245 std::swap(size, temp.size);
246
247 // 3. При выходе из if деструктор temp очистит старые ресурсы (которые теперь в temp).
248 }
249 return *this;
250}

263template<typename K, typename V>
264void HashTable<K, V>::rehash() {
265 Entry** old_buckets = buckets;
266 size_t old_bucket_count = bucket_count;
267
268 bucket_count *= 2;
269 buckets = new Entry*[bucket_count];
270 for (size_t i = 0; i < bucket_count; ++i) {
271 buckets[i] = nullptr;
272 }
273
274 size = 0;
275
276 for (size_t i = 0; i < old_bucket_count; ++i) {
277 Entry* current = old_buckets[i];
278 while (current) {
279 Entry* next = current->next;
280 size_t new_index = hash(current->key);
281 current->next = buckets[new_index];
282 buckets[new_index] = current;
283 ++size;
284 current = next;
285 }
286 }
287
288 delete[] old_buckets;
289}
