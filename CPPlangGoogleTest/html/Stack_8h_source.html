149template<typename T>
150Stack<T>::Stack(const Stack& other) : top_node(nullptr), size(0) {
151 if (other.top_node) {
152 // Создаем временный массив для инвертирования порядка элементов.
153 // Это необходимо, так как push добавляет в начало списка.
154 // Чтобы копия соответствовала оригиналу (Top -> Bottom), нужно вставлять элементы, начиная с самого нижнего.
155 T* temp = new T[other.size];
156 Node* current = other.top_node;
157 size_t index = other.size;
158 while (current) {
159 temp[--index] = current->data;
160 current = current->next;
161 }
162
163 // Вставляем элементы в правильном порядке
164 for (size_t i = 0; i < other.size; ++i) {
165 push(temp[i]);
166 }
167 delete[] temp;
168 }
169}

171template<typename T>
172Stack<T>& Stack<T>::operator=(const Stack& other) {
173 if (this != &other) {
174 // Идиома Copy-and-Swap.
175 // 1. Создаем копию через конструктор копирования.
176 // Если здесь произойдет исключение (например, нехватка памяти при new T[]),
177 // текущий объект останется нетронутым.
178 Stack<T> temp(other);
179
180 // 2. Обмениваем ресурсы.
181 std::swap(top_node, temp.top_node);
182 std::swap(size, temp.size);
183
184 // 3. Деструктор temp очистит старую память.
185 }
186 return *this;
